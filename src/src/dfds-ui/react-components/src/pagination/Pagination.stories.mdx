# Pagination

Pagination component

<Props of={Pagination} />

```jsx
import { Pagination } from '@/dfds-ui/react-components/src'
```

## Basic example

<Preview gray>
  <Story name="Basic">
    {() => {
      const [offset, setOffset] = React.useState(0)
      const [pageLimit, setPageLimit] = React.useState(15)
      const handlePageChange = (_event, page) => {
        setOffset(page * pageLimit - pageLimit)
      }
      const handlePageLimitChange = ({ value }) => {
        setPageLimit(value)
      }
      return (
        <Pagination
          offset={offset}
          onPageChange={handlePageChange}
          onChangePageLimit={handlePageLimitChange}
          options={[
            { label: '15 per page', value: 15 },
            { label: '30 per page', value: 30 },
            { label: '50 per page', value: 50 },
          ]}
          pageLimit={pageLimit}
          totalRecords={0}
        />
      )
    }}
  </Story>
</Preview>
